import asyncio
from js import document, fetch

async def main():
    # Obtener el contenido de hbd.txt
    response = await fetch("hbd.txt")
    text = await response.text()
    ascii_area = document.getElementById("ascii")
    for char in text:
        ascii_area.innerText += char
        await asyncio.sleep(0.005)  # velocidad de animaciÃ³n
asyncio.ensure_future(main())
