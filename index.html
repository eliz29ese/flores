import asyncio
from js import document, fetch

async def mostrar_ascii():
    response = await fetch("hbd.txt")
    text = await response.text()
    area = document.getElementById("ascii")
    for char in text:
        area.innerText += char
        await asyncio.sleep(0.005)

asyncio.ensure_future(mostrar_ascii())
